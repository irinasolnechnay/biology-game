  if(ok) score++;

  updateHeader();

 

  // îáúÿñíåíèå: â ýêçàìåíå îñòàâëÿåì «ñóõî», â òðåíèðîâêå  ìîæíî ìÿã÷å

  actions.style.display = "flex";
